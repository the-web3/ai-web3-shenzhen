pragma solidity ^0.8.0;

// 直接导入OpenZeppelin的IERC20接口，确保100%兼容
import "@openzeppelin/contracts/token/ERC20/IERC20.sol";

contract TinyDEX {
    address public owner;
    uint256 public constant PRICE = 10000;
    mapping(address => bool) public supportedTokens;
    
    event TokenBought(address buyer, address token, uint256 devAmount, uint256 tokenAmount);
    event TokenSold(address seller, address token, uint256 tokenAmount, uint256 devAmount);
    
    // 构造函数明确声明
    constructor() {
        owner = msg.sender;
    }
    
    function addSupportedToken(address tokenAddress) external {
        require(msg.sender == owner, "Only owner");
        supportedTokens[tokenAddress] = true;
    }
    
    function buyToken(address tokenAddress) external payable {
        require(supportedTokens[tokenAddress], "Token not supported");
        require(msg.value > 0, "Must send DEV");
        
        uint256 tokenAmount = msg.value * PRICE;
        require(
            IERC20(tokenAddress).balanceOf(address(this)) >= tokenAmount, 
            "Insufficient token liquidity"
        );
        
        require(
            IERC20(tokenAddress).transfer(msg.sender, tokenAmount),
            "Token transfer failed"
        );
        
        emit TokenBought(msg.sender, tokenAddress, msg.value, tokenAmount);
    }
    
    function sellToken(address tokenAddress, uint256 tokenAmount) external {
        require(supportedTokens[tokenAddress], "Token not supported");
        require(tokenAmount > 0, "Amount must be positive");
        
        uint256 devAmount = tokenAmount / PRICE;
        require(address(this).balance >= devAmount, "Insufficient DEV liquidity");
        
        require(
            IERC20(tokenAddress).transferFrom(msg.sender, address(this), tokenAmount),
            "Token transferFrom failed"
        );
        
        (bool success, ) = payable(msg.sender).call{value: devAmount}("");
        require(success, "DEV transfer failed");
        
        emit TokenSold(msg.sender, tokenAddress, tokenAmount, devAmount);
    }
    
    receive() external payable {}
}
